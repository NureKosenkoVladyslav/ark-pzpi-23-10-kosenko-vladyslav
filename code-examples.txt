def calculate_sum(first_number: int, second_number: int) -> int:
    """Повертає суму двох чисел."""
    return first_number + second_number

def main():
    number_a = 5
    number_b = 10
    print(f'Sum = {calculate_sum(number_a, number_b)}')

if __name__ == '__main__':
    main()
